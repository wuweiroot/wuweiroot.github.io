    优化#

    MessageSendTask 轮询机制低效：使用osDelay(1000)进行 1 秒轮询，可能导致命令响应延迟。#

    • 优化：改用事件驱动，通过消息队列或信号量直接唤醒任务。
    • 使用消息队列替代轮询，任务休眠直到有新命令到来，降低 CPU 使用率并提高响应速度。

    生产bin文件cmake配置#

    # 生成BIN文件
set(ELF_FILE ${PROJECT_NAME}.elf)
set(BIN_FILE ${PROJECT_NAME}.bin)

add_custom_command(
    TARGET ${PROJECT_NAME}
    POST_BUILD
    COMMAND ${CMAKE_OBJCOPY} -O binary ${ELF_FILE} ${BIN_FILE}
    COMMENT "Generating ${BIN_FILE}"
)

add_custom_target(
    bin
    COMMAND ${CMAKE_OBJCOPY} -O binary ${ELF_FILE} ${BIN_FILE}
    DEPENDS ${PROJECT_NAME}
    COMMENT "Generating ${BIN_FILE}"
)
